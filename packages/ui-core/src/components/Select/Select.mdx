# Select

## Properties

<Props of={Select} />

```javascript
const items = [
    {
        id: 1,
        value: "eat",
        title: "Eat"
    },
    {
        id: 2,
        value: "sleep",
        title: "Sleep"
    },
    {
        id: 3,
        value: "traing",
        title: "Traing"
    }
];
```

## Basic usage

<Playground>
    <SelectWrapper>
        {({ onSelectItem, selectedTitle, canOpen }) => (
            <Select
                selectedTitle={selectedTitle}
                onSelectItem={onSelectItem}
                placeholder="Choose your step"
                items={items}
            />
        )}
    </SelectWrapper>
</Playground>

## Search usage

<Playground>
    <SelectWrapper>
        {({ onSelectItem, onChangeSearch, searchValue, canOpen }) => (
            <Select
                placeholder="Choose your step"
                searchValue={searchValue}
                items={
                    searchValue
                        ? items.filter(
                              ({ title }) =>
                                  title
                                      .toLowerCase()
                                      .search(searchValue.toLowerCase()) !== -1
                          )
                        : []
                }
                onSelectItem={onSelectItem}
                onChangeSearch={onChangeSearch}
                arrow={false}
                canOpen={canOpen}
                label="Search"
                id="333"
            />
        )}
    </SelectWrapper>
</Playground>
