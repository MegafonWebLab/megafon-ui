export class SelectWrapper extends React.Component {
    constructor(props) {
        super(props);
        this.state = {
            selectedTitle: '',
            searchValue: '',
            canOpen: false,
        }
    }
    onSelectItem = (e, data) => {
        this.setState({
            selectedTitle: data.title,
            searchValue: data.title,
        })
    }
    onChangeSearch = value => {
        this.setState({
            searchValue: value,
            canOpen: !!value,
        });
    }
    render() {
        return this.props.children({
            onSelectItem: this.onSelectItem,
            onChangeSearch: this.onChangeSearch,
            selectedTitle: this.state.selectedTitle,
            searchValue: this.state.searchValue,
            canOpen: this.state.canOpen,
        })
    }
}

## Search usage

<Playground wrapper={Wrapper}>
    <SelectWrapper>
        {({ onSelectItem, onChangeSearch, searchValue, canOpen }) =>
                <Select
                    placeholder="Choose your step"
                    searchValue={searchValue}
                    items={
                        searchValue
                            ? items.filter(
                                ({ title }) =>
                                    title.toLowerCase().search(searchValue.toLowerCase()) !== -1,
                                )
                            : []
                    }
                    onSelectItem={onSelectItem}
                    onChangeSearch={onChangeSearch}
                    arrow={false}
                    canOpen={canOpen}
                    label="Search"
                />
        }
    </SelectWrapper>
</Playground>
