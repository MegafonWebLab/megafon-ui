    const [currentValue, setCurrentValue] = React.useState();

    const handleSelect = (_e, data) => {
        setCurrentValue(data.value);
    };

    return (
        <div style={DemoSelectWrapperStyle}>
            {props.children({
                onSelect: handleSelect,
                currentValue: currentValue,
            })}
        </div>
    );
};
