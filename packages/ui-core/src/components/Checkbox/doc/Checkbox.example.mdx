import { Playground, Props } from 'docz';
import Checkbox from '../Checkbox';
import Info from 'icons/System/16/Info_16.svg';
import { CheckboxWrapper, extraContentStyle } from './Checkbox.docz';

```javascript collapse=Код CheckboxWrapper
export const CheckboxWrapper: React.FC<ICheckboxWrapperProps> = (props) => {
    const [checked, setChecked] = React.useState<boolean>(false);

    const handleChange = () => {
        setChecked(!checked);
    };

    return (
        <div style={{ padding: '20px 10px 8px 10px' }}>
            {props.children({
                checked,
                onChange: handleChange,
            })}
        </div>
    );
};
```
