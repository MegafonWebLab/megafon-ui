
import { Playground } from 'docz';
import Partners from '../Partners';
import { ContentArea } from "@megafon/ui-core";
import { fourItems, eightItemsWithoutHref, twelveItems } from './Partners.docz.tsx'

## Особенности использования
Рекомендуется размещать четное количество партнеров и не меньше 4 штук

## Примеры свойств
``` javascript

const items = [
    {
        href: '#',
        src: 'image.png',
    },
    {
        src: 'image.png',
    },
]
```

# Базовое использование

<Playground>
    <ContentArea>
        <Partners items={fourItems} />
    </ContentArea>
</Playground>

# Логотипы без ссылок

<Playground>
    <ContentArea>
        <Partners items={eightItemsWithoutHref} />
    </ContentArea>
</Playground>

# Режим карусели
Режим карусели включается автоматически, если количество логотипов больше 8

<Playground>
    <ContentArea>
        <Partners items={twelveItems} />
    </ContentArea>
</Playground>
