import { Playground } from 'docz';
import Instructions from '../Instructions';
import {
    items,
    itemsWithVideo,
    iphoneInstructionItems,
    androidInstructionItems,
    newIphoneInstructionItems,
    laptopInstructionItems,

} from './Instructions.docz.tsx';

```javascript collapse=Демо данные
const items = [
    {
        title:  'Пункт инструкции №1';

    },
    {
        title: 'Пункт инструкции №2';
```

## Использование по умолчанию

<Playground style={wrapperStyles}>
    <Instructions title="Как подписаться" instructionItems={items} />
</Playground>

## С видео контентом

<Playground style={wrapperStyles}>
    <Instructions title="Как подписаться" instructionItems={itemsWithVideo} />
</Playground>

## Изображение справа

</Playground>
